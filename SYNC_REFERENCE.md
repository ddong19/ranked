## Supabase RPC Function (For Future Sync Implementation)

```sql
-- RPC function to atomically update item ranks within a ranking
CREATE OR REPLACE FUNCTION update_item_ranks(
  p_ranking_id INTEGER,
  p_item_ranks JSONB
) RETURNS void AS $$
DECLARE
  item_id INTEGER;
  new_rank INTEGER;
  max_rank INTEGER;
BEGIN
  -- Validate that ranking exists
  IF NOT EXISTS (SELECT 1 FROM ranking WHERE id = p_ranking_id) THEN
    RAISE EXCEPTION 'Ranking with id % does not exist', p_ranking_id;
  END IF;

  -- Get the maximum rank to use as offset
  SELECT COALESCE(MAX(rank), 0) + 1000 INTO max_rank 
  FROM item WHERE ranking_id = p_ranking_id;

  -- Step 1: Move all items to temporary high ranks to avoid constraint violations
  UPDATE item 
  SET rank = rank + max_rank 
  WHERE ranking_id = p_ranking_id;

  -- Step 2: Update items to their final ranks
  FOR item_id, new_rank IN
    SELECT (key)::INTEGER, (value)::INTEGER 
    FROM jsonb_each_text(p_item_ranks)
  LOOP
    -- Validate that the item belongs to the specified ranking
    IF NOT EXISTS (
      SELECT 1 FROM item 
      WHERE id = item_id AND ranking_id = p_ranking_id
    ) THEN
      RAISE EXCEPTION 'Item with id % does not belong to ranking %', item_id, p_ranking_id;
    END IF;

    -- Update the item's rank
    UPDATE item 
    SET rank = new_rank 
    WHERE id = item_id AND ranking_id = p_ranking_id;
  END LOOP;
END;
$$ LANGUAGE plpgsql;
```

## Database Schema (Supabase)

```sql
-- Rankings table
create table public.ranking (
  id bigint generated by default as identity not null,
  title character varying not null,
  description text null,
  constraint ranking_pkey primary key (id)
) TABLESPACE pg_default;

-- Items table  
create table public.item (
  id bigint generated by default as identity not null,
  name character varying not null,
  notes text null,
  rank bigint not null,
  ranking_id bigint not null,
  constraint item_pkey primary key (id),
  constraint item_rank_ranking_unique unique (rank, ranking_id),
  constraint fk_item_ranking foreign KEY (ranking_id) references ranking (id) on delete CASCADE
) TABLESPACE pg_default;

-- Trigger to adjust ranks after delete
create trigger trigger_adjust_ranks_after_delete
after DELETE on item for EACH row
execute FUNCTION adjust_ranks_after_delete ();
```
